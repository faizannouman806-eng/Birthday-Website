import React, { useState, useRef, useEffect } from 'react';
import { motion, AnimatePresence } from 'motion/react';
import { Sparkles, Heart, Flame, Music, Sparkle, ChevronRight } from 'lucide-react';
import { synth } from '../data';

interface BirthdayCakeProps {
  bestieName: string;
  onNextStage: () => void;
}

interface Particle {
  x: number;
  y: number;
  vx: number;
  vy: number;
  color: string;
  radius: number;
  rotation: number;
  rotationSpeed: number;
  opacity: number;
  fadeSpeed: number;
}

export default function BirthdayCake({ bestieName, onNextStage }: BirthdayCakeProps) {
  const [wishText, setWishText] = useState('');
  const [wishMade, setWishMade] = useState(false);
  const [candlesLit, setCandlesLit] = useState(true);
  const [blownAnimation, setBlownAnimation] = useState(false);
  const [musicActive, setMusicActive] = useState(false);

  const canvasRef = useRef<HTMLCanvasElement | null>(null);
  const particlesRef = useRef<Particle[]>([]);
  const animationFrameId = useRef<number | null>(null);

  // Colors for confetti
  const confettiColors = ['#ff7597', '#b026ff', '#3b82f6', '#10b981', '#f59e0b', '#ec4899', '#a855f7'];

  // Particle generation
  const triggerConfetti = () => {
    const canvas = canvasRef.current;
    if (!canvas) return;
    const ctx = canvas.getContext('2d');
    if (!ctx) return;

    const width = canvas.width;
    const height = canvas.height;

    // Create 180 sparkly particles
    for (let i = 0; i < 180; i++) {
      particlesRef.current.push({
        x: width / 2 + (Math.random() - 0.5) * 40,
        y: height * 0.55 + (Math.random() - 0.5) * 40, // originate from cake flame location
        vx: (Math.random() - 0.5) * 12,
        vy: -Math.random() * 16 - 4, // explode upwards
        color: confettiColors[Math.floor(Math.random() * confettiColors.length)],
        radius: Math.random() * 5 + 3,
        rotation: Math.random() * 360,
        rotationSpeed: (Math.random() - 0.5) * 10,
        opacity: 1,
        fadeSpeed: Math.random() * 0.008 + 0.005
      });
    }
  };

  // Canvas update loop
  useEffect(() => {
    const canvas = canvasRef.current;
    if (!canvas) return;

    const resizeCanvas = () => {
      canvas.width = canvas.parentElement?.clientWidth || window.innerWidth;
      canvas.height = canvas.parentElement?.clientHeight || 600;
    };

    resizeCanvas();
    window.addEventListener('resize', resizeCanvas);

    const ctx = canvas.getContext('2d');
    const update = () => {
      if (!ctx) return;
      ctx.clearRect(0, 0, canvas.width, canvas.height);

      const particles = particlesRef.current;
      for (let i = particles.length - 1; i >= 0; i--) {
        const p = particles[i];
        p.x += p.vx;
        p.y += p.vy;
        p.vy += 0.25; // gravity effect
        p.vx *= 0.98; // wind resistance
        p.rotation += p.rotationSpeed;
        p.opacity -= p.fadeSpeed;

        if (p.opacity <= 0 || p.y > canvas.height) {
          particles.splice(i, 1);
          continue;
        }

        ctx.save();
        ctx.translate(p.x, p.y);
        ctx.rotate((p.rotation * Math.PI) / 180);
        ctx.globalAlpha = p.opacity;
        ctx.fillStyle = p.color;

        // Draw diamond shapes or star shapes
        if (i % 2 === 0) {
          ctx.beginPath();
          ctx.moveTo(0, -p.radius);
          ctx.lineTo(p.radius * 0.6, 0);
          ctx.lineTo(0, p.radius);
          ctx.lineTo(-p.radius * 0.6, 0);
          ctx.closePath();
          ctx.fill();
        } else {
          ctx.fillRect(-p.radius / 2, -p.radius / 2, p.radius, p.radius);
        }
        ctx.restore();
      }

      animationFrameId.current = requestAnimationFrame(update);
    };

    update();

    return () => {
      window.removeEventListener('resize', resizeCanvas);
      if (animationFrameId.current) {
        cancelAnimationFrame(animationFrameId.current);
      }
    };
  }, []);

  const handleBlowOut = () => {
    if (!wishText.trim()) return;
    setWishMade(true);
    setCandlesLit(false);
    setBlownAnimation(true);

    // Stop previous music, play Happy Birthday song!
    synth.stop();
    synth.playHappyBirthday();
    setMusicActive(true);

    // Trigger multiple waves of sparkle explosion
    triggerConfetti();
    setTimeout(triggerConfetti, 400);
    setTimeout(triggerConfetti, 800);
  };

  return (
    <div className="relative min-h-screen bg-neutral-950 px-4 py-12 md:p-16 flex flex-col justify-between overflow-hidden">
      {/* Cosmos Background */}
      <div className="absolute inset-0 bg-radial-at-b from-slate-900 via-neutral-950 to-black pointer-events-none" />
      
      {/* Sparkles Canvas overlay */}
      <canvas 
        ref={canvasRef} 
        className="absolute inset-0 z-20 pointer-events-none"
      />

      {/* Nebula ambient gradients */}
      <div className="absolute top-1/3 left-1/2 -translate-x-1/2 w-96 h-96 rounded-full bg-pink-500/10 blur-3xl pointer-events-none" />

      {/* Header */}
      <div className="relative z-10 text-center max-w-2xl mx-auto">
        <div className="inline-flex items-center gap-1 bg-white/5 border border-white/10 px-4 py-1.5 rounded-full text-xs text-rose-300 font-mono tracking-widest uppercase mb-4">
          <Sparkles className="w-3.5 h-3.5 text-rose-glow" />
          <span>Stage II • The Wishing Well</span>
        </div>
        <h2 className="font-serif text-3xl md:text-5xl font-light text-white tracking-wide">
          The Virtual <span className="font-cursive text-rose-glow text-4xl md:text-6xl">Wishing Well</span>
        </h2>
        <p className="mt-3 text-xs md:text-sm text-gray-400 font-sans tracking-wide max-w-md mx-auto">
          Every special year deserves a special dream. Type your secret birthday wish below, then blow out the virtual candles to unlock your magical birthday letter!
        </p>
      </div>

      {/* Wishing area & Animated Cake */}
      <div className="relative z-10 w-full max-w-3xl mx-auto flex flex-col md:flex-row items-center justify-center gap-12 my-8">
        
        {/* Animated CSS Cake Panel */}
        <div className="relative flex flex-col items-center justify-center w-full max-w-[280px] h-[320px]" id="cake-wrapper">
          
          {/* Cake Stand */}
          <div className="absolute bottom-6 w-52 h-3.5 bg-neutral-800 border-b border-neutral-700 rounded-full shadow-lg z-0" />
          <div className="absolute bottom-1 w-12 h-6 bg-neutral-700 rounded-t-lg z-0" />

          {/* Layer 1: Bottom Layer */}
          <motion.div 
            animate={{ y: [0, -3, 0] }}
            transition={{ repeat: Infinity, duration: 4, ease: 'easeInOut' }}
            className="absolute bottom-9 w-44 h-18 bg-rose-950 border border-rose-900/40 rounded-t-2xl shadow-xl z-10 flex flex-col justify-end overflow-hidden"
          >
            {/* Dripping frosting effect */}
            <div className="w-full flex justify-between px-0.5">
              {[...Array(6)].map((_, i) => (
                <div key={i} className="w-6 h-4 bg-pink-500/70 rounded-b-full shadow-inner" />
              ))}
            </div>
          </motion.div>

          {/* Layer 2: Top Layer */}
          <motion.div 
            animate={{ y: [0, -4, 0] }}
            transition={{ repeat: Infinity, duration: 4, delay: 0.5, ease: 'easeInOut' }}
            className="absolute bottom-24 w-34 h-16 bg-neutral-900 border border-rose-900/30 rounded-t-xl shadow-lg z-20 flex flex-col justify-end overflow-hidden"
          >
            {/* White/Pink piping cream dots */}
            <div className="w-full flex justify-between px-1">
              {[...Array(5)].map((_, i) => (
                <div key={i} className="w-4 h-3.5 bg-rose-400/80 rounded-b-full shadow" />
              ))}
            </div>
          </motion.div>

          {/* Birthday Candles */}
          <div className="absolute bottom-38 w-full flex justify-center gap-8 z-30">
            {[...Array(3)].map((_, idx) => (
              <div key={idx} className="relative flex flex-col items-center">
                {/* Candle Flame with Glow */}
                <AnimatePresence>
                  {candlesLit && (
                    <motion.div
                      initial={{ scale: 0, opacity: 0 }}
                      animate={{ 
                        scale: [1, 1.15, 1],
                        opacity: 1,
                        y: [0, -2, 0]
                      }}
                      exit={{ 
                        scale: 0, 
                        opacity: 0,
                        transition: { duration: 0.4 }
                      }}
                      transition={{ 
                        repeat: Infinity, 
                        duration: 1.5 + idx * 0.2, 
                        ease: 'easeInOut' 
                      }}
                      className="absolute -top-7 w-4.5 h-6 bg-gradient-to-t from-yellow-500 via-orange-500 to-rose-500 rounded-full blur-[1px] shadow-[0_0_12px_#f97316] flex items-center justify-center cursor-pointer"
                    >
                      <Flame className="w-2.5 h-2.5 text-white fill-current opacity-80" />
                    </motion.div>
                  )}
                </AnimatePresence>

                {/* Candle Body */}
                <div className={`w-3.5 h-10 bg-gradient-to-b ${idx === 1 ? 'from-violet-400 to-violet-600' : 'from-pink-400 to-pink-600'} border border-white/10 rounded-t-sm shadow`} />
              </div>
            ))}
          </div>

          {/* Sparkly text status indicator */}
          <div className="absolute top-10 font-mono text-[10px] text-gray-500 uppercase tracking-widest text-center">
            {candlesLit ? '🔥 Candles are burning' : '✨ Wish Released Into Star Dust'}
          </div>
        </div>

        {/* Wishes Input Card */}
        <motion.div 
          initial={{ opacity: 0, x: 20 }}
          animate={{ opacity: 1, x: 0 }}
          transition={{ duration: 1 }}
          className="w-full max-w-sm bg-neutral-900/40 border border-white/10 p-6 md:p-8 rounded-3xl backdrop-blur-xl relative z-30"
          id="wish-control-card"
        >
          {!wishMade ? (
            <>
              <h3 className="font-serif text-xl text-white font-medium flex items-center gap-2">
                <Heart className="w-5 h-5 text-rose-glow animate-pulse fill-current" />
                <span>Make a Wish</span>
              </h3>
              <p className="text-xs text-gray-400 mt-2 font-sans leading-relaxed">
                What does your heart desire this year? Type it into the star-catcher. It is strictly private between us.
              </p>

              <textarea
                value={wishText}
                onChange={(e) => setWishText(e.target.value)}
                placeholder={`I wish for endless coffee dates, beautiful chapters, and...`}
                className="w-full bg-neutral-950/60 border border-white/10 rounded-2xl py-3.5 px-4 mt-4 text-sm text-white placeholder-gray-500 focus:outline-none focus:border-rose-400/50 focus:ring-1 focus:ring-rose-400/20 transition-all font-sans h-24 resize-none"
                id="birthday-wish-textarea"
              />

              <button
                onClick={handleBlowOut}
                disabled={!wishText.trim()}
                className={`w-full py-4 rounded-xl font-sans text-xs font-semibold tracking-widest mt-4 transition-all flex items-center justify-center gap-2 cursor-pointer shadow-lg ${
                  wishText.trim()
                    ? 'bg-gradient-to-r from-rose-500 to-pink-600 text-white shadow-pink-500/10'
                    : 'bg-neutral-800 text-gray-500 border border-white/5 cursor-not-allowed'
                }`}
                id="blow-candles-btn"
              >
                <span>BLOW OUT CANDLES 🎂</span>
              </button>
            </>
          ) : (
            <motion.div 
              initial={{ opacity: 0, scale: 0.95 }}
              animate={{ opacity: 1, scale: 1 }}
              className="text-center py-6"
            >
              <div className="inline-flex items-center justify-center bg-green-500/10 border border-green-500/20 p-4 rounded-full mb-4 text-green-400">
                <Sparkle className="w-8 h-8 animate-spin" />
              </div>
              <h3 className="font-serif text-xl text-white font-medium">
                Your Wish is Sent!
              </h3>
              <p className="text-xs text-gray-400 mt-2 font-sans leading-relaxed px-2">
                The virtual candles have been extinguished, {bestieName}. May every letter of your wish echo through the cosmos and bloom in your life.
              </p>

              {musicActive && (
                <div className="mt-4 inline-flex items-center gap-2 bg-rose-500/10 border border-rose-500/25 px-4 py-1.5 rounded-full text-rose-glow text-xs font-mono">
                  <Music className="w-3.5 h-3.5 animate-bounce" />
                  <span>PLAYING BIRTHDAY SYNTH</span>
                </div>
              )}
            </motion.div>
          )}
        </motion.div>

      </div>

      {/* Stage Controller Button */}
      <div className="relative z-10 flex justify-center mt-4">
        <motion.button
          onClick={onNextStage}
          disabled={!wishMade}
          className={`flex items-center gap-2 px-8 py-4.5 rounded-full font-sans font-medium tracking-widest text-xs transition-all shadow-lg ${
            wishMade
              ? 'bg-gradient-to-r from-rose-500 to-violet-600 hover:opacity-95 text-white shadow-pink-500/10 cursor-pointer'
              : 'bg-neutral-900 text-gray-600 border border-white/5 cursor-not-allowed'
          }`}
          whileHover={wishMade ? { scale: 1.03 } : {}}
          whileTap={wishMade ? { scale: 0.98 } : {}}
          id="stage-next-cake-btn"
        >
          <span>READ YOUR BIRTHDAY LETTERS</span>
          <ChevronRight className="w-4 h-4" />
        </motion.button>
      </div>
    </div>
  );
}
